
1. the assign destination must be a left-value

int fn() {
    1 = 2;
    return 0;
}


2. the assign destination cannot be constant

void f(char text[], char ch) {
    text = ch;
}



4. the assign source must be scalar

int fn() {
    int x;
    int v[5];
    x = v;
    return 0;
}


5. the assign source cannot be converted to destination

struct A { int x; };
struct B { int y; };
int fn() {
    struct A a;
    struct B b;
    a = b;
    return 0;
}


6. the if condition must be a scalar value

int fn() {
    int v[5];
    if(v) {}
    return 0;
}


7. the while condition must be a scalar value

int fn() {
    int v[5];
    while(v) {}
    return 0;
}


8. a void function cannot return a value

void fn() {
    return 1;
}


9. the return value must be a scalar value

int fn() {
    int v[5];
    return v;
}


10. cannot convert the return expression type to the function return type

struct A { int x; };
struct B { int y; };
struct A fn() {
    struct B b;
    return b;
}


11. a non-void function must return a value

int fn() {
    return;
}


12. invalid operand type for ||

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a || b;
    return 0;
}


13. invalid operand type for &&

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a && b;
    return 0;
}


14. invalid operand type for == or !=

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a == b;
    return 0;
}


15. invalid operand type for <, <=, >, >=

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a < b;
    return 0;
}


16. invalid operand type for + or -

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a + b;
    return 0;
}


17. invalid operand type for * or /

struct A { int x; };
int fn() {
    struct A a;
    struct A b;
    int x;
    x = a * b;
    return 0;
}


18. cannot convert to a struct type

struct A { int x; };
int fn() {
    int x;
    x = (struct A)1;
    return 0;
}


19. cannot convert a struct

struct A { int x; };
int fn() {
    struct A a;
    int x;
    x = (int)a;
    return 0;
}


20. an array can be converted only to another array

int fn() {
    int v[5];
    int x;
    x = (int)v;
    return 0;
}


21. a scalar can be converted only to another scalar

int fn() {
    int x;
    x = (int[5])x;
    return 0;
}


22. unary - or ! must have a scalar operand

int fn() {
    int v[5];
    int x;
    x = -v;
    return 0;
}


23. only an array can be indexed

int fn() {
    int x;
    x[0] = 1;
    return 0;
}


24. the index is not convertible to int

struct A { int x; };
int fn() {
    int v[5];
    struct A a;
    v[a] = 1;
    return 0;
}


25. a field can only be selected from a struct

int fn() {
    int x;
    x.y = 1;
    return 0;
}


26. the structure does not have a field

struct A { int x; };
int fn() {
    struct A a;
    a.z = 1;
    return 0;
}


27. undefined id

int fn() {
    x = 1;
    return 0;
}


28. only a function can be called

int fn() {
    int x;
    x();
    return 0;
}


29. too many arguments in function call

int fn(int a) {
    return 0;
}
int h() {
    fn(1, 2);
    return 0;
}


30. too few arguments in function call

int fn(int a) {
    return 0;
}
int h() {
    fn();
    return 0;
}


31. in call, cannot convert the argument type to the parameter type

struct A { int x; };
void fn(int a) {}
int h() {
    struct A a;
    fn(a);
    return 0;
}


32. a function can only be called

int fn() {
    return 0;
}
int h() {
    int x;
    x = fn;
    return 0;
}
